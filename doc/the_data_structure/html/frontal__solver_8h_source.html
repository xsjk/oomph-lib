145 /// Clean up memory
146 void clean_up_memory()
147 {
148 if (IW)
149 {
150 delete[] IW;
151 IW = 0;
152 Liw = 0;
153 }
154 if (W)
155 {
156 delete[] W;
157 W = 0;
158 Lw = 0;
159 }
160 }

162 /// Overload disable resolve so that it cleans up memory too
163 void disable_resolve()

166 clean_up_memory();
167 }
