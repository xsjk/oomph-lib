69 class SamplePointContainerParameters

86 void operator=(const SamplePointContainerParameters&) = delete;
87
88 /// Empty destructor
89 virtual ~SamplePointContainerParameters() {}
90
91 /// Pointer to mesh from whose FiniteElements sample points are created
92 Mesh* mesh_pt() const

101 Vector<std::pair<double, double>>& min_and_max_coordinates()

103 return Min_and_max_coordinates;
104 }

110 Vector<std::pair<double, double>> min_and_max_coordinates() const

112 return Min_and_max_coordinates;

114

120 }
121

124 {
125 return Nsample_points_generated_per_element;
126 }
127
128 /// Use eulerian coordinates (via interpolated_x) during
129 /// setup (otherwise use interpolated_zeta())?
130 bool use_eulerian_coordinates_during_setup() const
131 {
132 return Use_eulerian_coordinates_during_setup;
133 }
134
135 /// Enable use of eulerian coordinates (via interpolated_x) during
136 /// setup (otherwise use interpolated_zeta())
137 void enable_use_eulerian_coordinates_during_setup()
138 {
139 Use_eulerian_coordinates_during_setup = true;
140 }
141
142 /// Disable use of eulerian coordinates (via interpolated_x) during
143 /// setup (otherwise use interpolated_zeta())
144 void disable_use_eulerian_coordinates_during_setup()
145 {
146 Use_eulerian_coordinates_during_setup = false;
147 }
148
149 /// Ignore halo elements? (MPI only)
150 bool ignore_halo_elements_during_locate_zeta_search() const
151 {
152 return Ignore_halo_elements_during_locate_zeta_search;
153 }
154
155 /// Enable Ignore halo elements? (MPI only)
156 void enable_ignore_halo_elements_during_locate_zeta_search()
157 {
158 Ignore_halo_elements_during_locate_zeta_search = true;
159 }
160
161 /// Disable Ignore halo elements? (MPI only)
162 void disable_ignore_halo_elements_during_locate_zeta_search()
163 {
164 Ignore_halo_elements_during_locate_zeta_search = false;
165 }
166
167 /// Default for "measure of" number of sample points per element
168 static unsigned Default_nsample_points_generated_per_element;
169
170 protected:
171 /// Pointer to mesh from whose FiniteElements sample points are created
172 Mesh* Mesh_pt;
173
174 /// Vector of pairs of doubles for min and maximum coordinates.
175 /// Call: Min_and_max_coordinates[j] gives me the
176 /// pair of min (first) and max. (second) coordinates in the j-th
177 /// coordinate direction.
178 Vector<std::pair<double, double>> Min_and_max_coordinates;
179
180 /// "Measure of" number of sample points generated in each element
181 unsigned Nsample_points_generated_per_element;
182
183 /// Use Eulerian coordinates to setup bin (i.e. use interpolated_x()
184 /// rather than interpolated_zeta() when setting up and searching sample
185 /// point container)
186 bool Use_eulerian_coordinates_during_setup;
187
188 /// Ignore halo elements? Accepting halo elements can drastically
189 /// reduce the number of external halo elements in multidomain
190 /// problems -- currently not aware of any problems with doing this
191 /// therefore set to false by default but retention
192 /// of this flag allows easy return to previous implementation.
193 bool Ignore_halo_elements_during_locate_zeta_search;
194
195
196 private:
197 /// Constructor: Pass mesh.
198 /// Constructor is private and can only be called
199 /// by the derived friends.
200 SamplePointContainerParameters(Mesh* mesh_pt)
201 : Mesh_pt(mesh_pt),
202 Nsample_points_generated_per_element(
203 Default_nsample_points_generated_per_element),
204 Use_eulerian_coordinates_during_setup(false),
205 Ignore_halo_elements_during_locate_zeta_search(false)
206 {
207 }
208
209 /// Broken default constructor; needed for broken
210 /// copy constructors. Don't call. It will die.
211 SamplePointContainerParameters()
212 {
213 // Throw the error
214 throw OomphLibError("Broken default constructor. Don't call this!",
215 OOMPH_CURRENT_FUNCTION,
216 OOMPH_EXCEPTION_LOCATION);
217 }
218 };
