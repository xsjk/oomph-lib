82/// Global variables
83//================================================================
84namespace Global_Parameters
85{
86
87 /// Poisson's ratio
88 double Nu=0.3;
89
90 /// Create constitutive law
91 ConstitutiveLaw* Constitutive_law_pt=new GeneralisedHookean(&Nu);
92
93 /// Non-dim gravity
94 double Gravity=0.0;
95
96 /// Non-dimensional gravity as body force
97 void gravity(const double& time,
98 const Vector<double> &xi,
99 Vector<double> &b)
100 {
101 b[0]=-Gravity;
102 b[1]=0.0;
103 b[2]=0.0;
104 } // end gravity
105
106 /// Uniform pressure
107 double P = 0.0;
108
109 /// Constant pressure load. The arguments to this function are imposed
110 /// on us by the SolidTractionElements which allow the traction to
111 /// depend on the Lagrangian and Eulerian coordinates x and xi, and on the
112 /// outer unit normal to the surface. Here we only need the outer unit
113 /// normal.
114 void constant_pressure(const Vector<double> &xi, const Vector<double> &x,
115 const Vector<double> &n, Vector<double> &traction)
116 {
117 unsigned dim = traction.size();
118 for(unsigned i=0;i<dim;i++)
119 {
120 traction[i] = -P*n[i];
121 }
122 } // end traction

125} //end namespace

177//=============start_constructor==========================================
178/// Constructor for unstructured solid problem
179//========================================================================
180template<class ELEMENT>
181UnstructuredSolidProblem<ELEMENT>::UnstructuredSolidProblem()
182{
183
184 //Create solid bulk mesh
185 string node_file_name="solid_iliac.1.node";
186 string element_file_name="solid_iliac.1.ele";
187 string face_file_name="solid_iliac.1.face";
188 Solid_mesh_pt = new MySolidTetgenMesh<ELEMENT>(node_file_name,
189 element_file_name,
190 face_file_name);
191
192 // The following IDs corresponds to the boundary IDs specified in
193 // the *.poly file from which tetgen generated the unstructured mesh.
194
195 /// IDs of solid mesh boundaries where displacements are pinned
196 Pinned_solid_boundary_id.resize(42);
197 for(unsigned i=0; i<42; i++)
198 {
199 Pinned_solid_boundary_id[i]=215+i;
200 }
201
202 // The solid mesh boundaries where an internal pressure is applied
203 Solid_traction_boundary_id.resize(215);
204 for(unsigned i=0; i<215; i++)
205 {
206 Solid_traction_boundary_id[i]=i;
207 }
208
209
210
211 // Apply BCs for solid
212 //--------------------
213
214 // Doc pinned solid nodes
215 std::ofstream bc_file("RESLT/pinned_solid_nodes.dat");
216
217 // Pin positions at inflow boundary (boundaries 0 and 1)
218 unsigned n=Pinned_solid_boundary_id.size();
219 for (unsigned i=0;i<n;i++)
220 {
221 // Get boundary ID
222 unsigned b=Pinned_solid_boundary_id[i];
223 unsigned num_nod= Solid_mesh_pt->nboundary_node(b);
224 for (unsigned inod=0;inod<num_nod;inod++)
225 {
226 // Get node
227 SolidNode* nod_pt=Solid_mesh_pt->boundary_node_pt(b,inod);
228
229 // Pin all directions
230 for (unsigned i=0;i<3;i++)
231 {
232 nod_pt->pin_position(i);
233
234 // ...and doc it as pinned
235 bc_file << nod_pt->x(i) << " ";
236 }
237
238 bc_file << std::endl;
239 }
240 }
241 bc_file.close();
242
243
244
245 // Complete the build of all elements so they are fully functional
246 //----------------------------------------------------------------
247 unsigned n_element = Solid_mesh_pt->nelement();
248 for(unsigned i=0;i<n_element;i++)
249 {
250 //Cast to a solid element
251 ELEMENT *el_pt = dynamic_cast<ELEMENT*>(
252 Solid_mesh_pt->element_pt(i));
253
254 // Set the constitutive law
255 el_pt->constitutive_law_pt() =
256 Global_Parameters::Constitutive_law_pt;
257
258 //Set the body force
259 el_pt->body_force_fct_pt() = Global_Parameters::gravity;
260 }
261
262
263 // Create traction elements
264 //-------------------------
265
266 // Create meshes of traction elements
267 n=Solid_traction_boundary_id.size();
268 Solid_traction_mesh_pt.resize(n);
269 for (unsigned i=0;i<n;i++)
270 {
271 Solid_traction_mesh_pt[i]=new SolidMesh;
272 }
273
274 // Build the traction elements
275 create_traction_elements();
276
277
278 // Combine the lot
279 //----------------
280
281 // The solid bulk mesh
282 add_sub_mesh(Solid_mesh_pt);
283
284 // The solid traction meshes
285 n=Solid_traction_boundary_id.size();
286 for (unsigned i=0;i<n;i++)
287 {
288 add_sub_mesh(Solid_traction_mesh_pt[i]);
289 }
290
291 // Build global mesh
292 build_global_mesh();
293
294 // Setup equation numbering scheme
295 std::cout <<"Number of equations: " << assign_eqn_numbers() << std::endl;
296
297} // end constructor

301//============start_of_create_traction_elements==========================
302/// Create traction elements
303//=======================================================================
304template<class ELEMENT>
305void UnstructuredSolidProblem<ELEMENT>::create_traction_elements()
306{
307
308 // Loop over traction boundaries
309 unsigned n=Solid_traction_boundary_id.size();
310 for (unsigned i=0;i<n;i++)
311 {
312 // Get boundary ID
313 unsigned b=Solid_traction_boundary_id[i];
314
315 // How many bulk elements are adjacent to boundary b?
316 unsigned n_element = Solid_mesh_pt->nboundary_element(b);
317
318 // Loop over the bulk elements adjacent to boundary b
319 for(unsigned e=0;e<n_element;e++)
320 {
321 // Get pointer to the bulk element that is adjacent to boundary b
322 ELEMENT* bulk_elem_pt = dynamic_cast<ELEMENT*>(
323 Solid_mesh_pt->boundary_element_pt(b,e));
324
325 //What is the index of the face of the element e along boundary b
326 int face_index = Solid_mesh_pt->face_index_at_boundary(b,e);
327
328 // Create new element
329 SolidTractionElement<ELEMENT>* el_pt=
330 new SolidTractionElement<ELEMENT>(bulk_elem_pt,face_index);
331
332 // Add it to the mesh
333 Solid_traction_mesh_pt[i]->add_element_pt(el_pt);
334
335 //Set the traction function
336 el_pt->traction_fct_pt() = Global_Parameters::constant_pressure;
337 }
338 }
339
340} // end of create_traction_elements

385//============================start_main==================================
386/// Demonstrate how to solve an unstructured 3D solid problem
387//========================================================================
388int main(int argc, char **argv)
389{
390 // Store command line arguments
391 CommandLineArgs::setup(argc,argv);
392
393 // Label for output
394 DocInfo doc_info;
395
396 // Output directory
397 doc_info.set_directory("RESLT");
398
399 //Set up the problem
400 UnstructuredSolidProblem<TPVDElement<3,3> > problem;
401
402 //Output initial configuration
403 problem.doc_solution(doc_info);
404 doc_info.number()++;
405
406 // Parameter study
407 Global_Parameters::P=0.0;
408 double g_increment=0.0;
409 double p_increment=1.0e-2;
410
411 unsigned nstep=4;
412 if (CommandLineArgs::Argc!=1)
413 {
414 std::cout << "Validation -- only doing two steps" << std::endl;
415 nstep=2;
416 }
417
418 // Do the parameter study
419 for (unsigned istep=0;istep<nstep;istep++)
420 {
421 // Solve the problem
422 problem.newton_solve();
423
424 //Output solution
425 problem.doc_solution(doc_info);
426 doc_info.number()++;
427
428 // Bump up load
429 Global_Parameters::Gravity+=g_increment;
430 Global_Parameters::P+=p_increment;
431
432 }
433
434} // end main
